144static inline ITCM_FUNC void __init_pipe(uint8_t message_pipe_idx, uint8_t max_messages,
145 bool should_engage)
146{
147 message_pipe_list[message_pipe_idx]->count = 0;

155 message_pipe_list[message_pipe_idx]->buffer[i] = 0;

212ITCM_FUNC void __os_init(void)
213{
214 uint8_t i;
215
216 os_running = 0;
217 ticks_per_task = ICARUS_TICKS_PER_TASK;
218 running_task_count = 0;
219 current_task_index = 0;
220 current_cleanup_task_idx = -1;
221 current_task_ticks_remaining = ticks_per_task;
222
223 for (i = 0; i < ICARUS_MAX_TASKS; i++) {
224 task_list[i] = &task_pool[i];
225 cleanup_task_idx[i] = -1;
226 data_pool_word_offsets[i] = 0;
227 }
228
229 for (i = 0; i < ICARUS_MAX_SEMAPHORES; i++) {
230 semaphore_list[i] = &semaphore_pool[i];
231 __init_sem(i, 0, false);
232 }
233
234 for (i = 0; i < ICARUS_MAX_MESSAGE_QUEUES; i++) {
235 message_pipe_list[i] = &message_pipe_pool[i];
236 __init_pipe(i, 0, false);
237 }
238
239 for (i = 0; i < 16; i++) {
240 cpu_vregisters[i] = 0;
241 }
242
243 __os_register_task(os_idle_task, "ICARUS_KEEPALIVE_TASK");
244 // __os_register_task(os_heartbeat_task, ">ICARUS_HEARTBEAT<"); // Disabled for interactive demo
245
246 scheduler_enabled = true;
247}

295ITCM_FUNC void* __kernel_protected_data(uint16_t num_words) {
296 if (data_pool_word_offsets[current_task_index] + num_words > ICARUS_DATA_WORDS || num_words == 0)
297 return NULL;
298 uint16_t current_offset = data_pool_word_offsets[current_task_index];
299 data_pool_word_offsets[current_task_index] += num_words;
300 uint32_t *ret_ptr = &data_pool[current_task_index][current_offset];
301 return (void*) ret_ptr;
302}

314ITCM_FUNC const char* __os_get_task_name(uint8_t task_idx) {
315 /* Static buffer in RAM_D1 (not DTCM) for unprivileged access */
316#ifndef HOST_TEST
317 static char name_buffer[ICARUS_MAX_TASK_NAME_LEN] __attribute__((section(".ram_d1")));
318#else
319 static char name_buffer[ICARUS_MAX_TASK_NAME_LEN];
320#endif
321
322 if (task_idx >= num_created_tasks || task_list[task_idx] == NULL) {
323 return NULL;
324 }
325
326 /* Copy name from DTCM to RAM_D1 buffer */
327 const char* src = task_list[task_idx]->name;
328 for (uint8_t i = 0; i < ICARUS_MAX_TASK_NAME_LEN; i++) {
329 name_buffer[i] = src[i];
330 if (src[i] == '\0') break;
331 }
332 name_buffer[ICARUS_MAX_TASK_NAME_LEN - 1] = '\0'; /* Ensure null termination */
333
334 return name_buffer;
335}
