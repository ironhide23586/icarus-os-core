43/* ============================================================================
44 * os_yield_pendsv - PendSV Context Switch Handler
45 * ============================================================================
46 *
47 * Called from PendSV_Handler to perform context switch.
48 * Saves current task context, selects next ready task, restores its context.
49 *
50 * Register usage:
51 * r0 - current_task_index address / scratch
52 * r1 - task_list address / next task SP
53 * r2 - scratch (index calculations)
54 * r3 - current TCB pointer
55 * r4 - current PSP / task state
56 * r5 - num_created_tasks
57 * r6 - scheduler_enabled / scratch
58 * r7 - os_tick_count / scratch
59 * r8 - scratch (tick calculations)
60 * r9 - next_task_index
61 * r10 - scratch (index * 4)
62 * r11 - next TCB pointer
63 */
64
65.global os_yield_pendsv
66.type os_yield_pendsv, %function
67
68os_yield_pendsv:
69 /* Save current task context */
70 mrs r0, psp /* Read Process Stack Pointer */
71 stmdb r0!, {r4-r11} /* Push R4-R11 to task stack */
72 mov r4, r0 /* Save updated PSP */
73
74 /* Load kernel state pointers */
75 ldr r0, =current_task_index
76 ldr r1, =task_list
77 ldr r5, =num_created_tasks
78 ldrb r5, [r5] /* r5 = task count */
79 ldrb r9, [r0] /* r9 = current index (will become next) */
80
81 /* Save current task's stack pointer to TCB */
82 ldrb r2, [r0] /* r2 = current_task_index */
83 lsl r2, r2, #2 /* r2 = index * 4 (pointer size) */
84 ldr r3, [r1, r2] /* r3 = task_list[current_task_index] */
85 str r4, [r3, #TCB_STACK_PTR] /* Save PSP to TCB */
86
87 /* Check if current task should transition to READY */
88 ldrb r4, [r3, #TCB_TASK_STATE]
89 cmp r4, #STATE_RUNNING
90 beq set_task_to_ready
91
92find_next_task:
93 /* Round-robin: next = (current + 1) % num_tasks */
94 mov r6, r9
95 add r6, r6, #1
96 udiv r7, r6, r5 /* r7 = (current+1) / num_tasks */
97 mul r8, r7, r5 /* r8 = quotient * num_tasks */
98 sub r9, r6, r8 /* r9 = next_task_index (remainder) */
99
100 /* Load next task's TCB */
101 lsl r10, r9, #2 /* r10 = next_index * 4 */
102 ldr r11, [r1, r10] /* r11 = task_list[next_index] */
103 ldrb r4, [r11, #TCB_TASK_STATE] /* r4 = next task state */
104
105 /* State machine: determine if task is runnable */
106 cmp r4, #STATE_COLD
107 beq yield_postprocess /* COLD tasks can start */
108
109 cmp r4, #STATE_READY
110 beq yield_postprocess /* READY tasks can run */
111
112 cmp r4, #STATE_KILLED
113 beq find_next_task /* Skip KILLED tasks */
114
115 cmp r4, #STATE_FINISHED
116 beq find_next_task /* Skip FINISHED tasks */
117
118 cmp r4, #STATE_BLOCKED
119 beq check_sleep_ticks /* Check if sleep expired */
120
121 b find_next_task /* Unknown state, skip */
122
123yield_postprocess:
124 /* Switch to next task */
125 strb r9, [r0] /* Update current_task_index */
126 ldr r1, [r11, #TCB_STACK_PTR] /* Load next task's SP */
127 mov r5, #STATE_RUNNING
128 strb r5, [r11, #TCB_TASK_STATE] /* Set state to RUNNING */
129
130 /* Check if task is cold (first run) */
131 cmp r4, #STATE_COLD
132 beq increment_running_count
133
134 /* Warm task: restore software-saved registers */
135 ldmia r1!, {r4-r11} /* Pop R4-R11 from task stack */
136
137branch_to_next_task:
138 msr psp, r1 /* Switch to next task's stack */
139 bx lr /* Return from exception */
140
141check_sleep_ticks:
142 /* Check if blocked task's sleep has expired */
143 ldr r7, =os_tick_count
144 ldr r7, [r7] /* r7 = current tick */
145 ldr r6, [r11, #TCB_TICK_PAUSED] /* r6 = tick when blocked */
146 ldr r8, [r11, #TCB_TICKS_PAUSE] /* r8 = ticks to sleep */
147 sub r10, r7, r6 /* r10 = elapsed ticks */
148 cmp r8, r10 /* ticks_to_pause vs elapsed */
149 bhs find_next_task /* Still sleeping, try next */
150 b yield_postprocess /* Sleep done, run this task */
151
152set_task_to_ready:
153 /* Check if scheduler is enabled before transitioning */
154 ldr r6, =scheduler_enabled
155 ldrb r6, [r6]
156 cmp r6, #0
157 beq defer_ctx_switch /* Scheduler disabled, defer */
158
159 mov r4, #STATE_READY
160 strb r4, [r3, #TCB_TASK_STATE] /* Set current to READY */
161 b find_next_task
162
163increment_running_count:
164 /* First time running this task */
165 ldr r7, =running_task_count
166 ldrb r6, [r7]
167 add r6, r6, #1
168 strb r6, [r7]
169 b branch_to_next_task
170
171defer_ctx_switch:
172 /* Scheduler disabled: re-run current task */
173 mov r11, r3 /* Next = current */
174 b yield_postprocess
175

178/* ============================================================================
179 * start_cold_task - Initialize and Start First Task
180 * ============================================================================
181 *
182 * Called once from os_start() to begin task execution.
183 * Sets up PSP, enables interrupts, and jumps to first task.
184 *
185 * @param r0 Pointer to task control block (TCB)
186 *
187 * Stack frame expected (set up by os_create_task):
188 * [SP+24] xPSR
189 * [SP+20] PC (entry point)
190 * [SP+16] LR (os_exit_task)
191 * [SP+12] R12
192 * [SP+8] R3
193 * [SP+4] R2
194 * [SP+0] R1
195 * [SP-4] R0 <- Initial SP points here
196 */
197
198.global start_cold_task
199.type start_cold_task, %function
200
201start_cold_task:
202 /* Load task's initial stack pointer */
203 ldr r1, [r0, #TCB_STACK_PTR]
204 msr psp, r1 /* Set Process Stack Pointer */
205
206 /* Switch to PSP for thread mode */
207 mov r1, #0x02 /* SPSEL = 1 (use PSP) */
208 msr control, r1
209 isb /* Instruction barrier */
210
211 /* Mark task as RUNNING */
212 mov r1, #STATE_RUNNING
213 strb r1, [r0, #TCB_TASK_STATE]
214
215 /* Pop exception frame from stack */
216 pop {r0-r3, r12, lr} /* R0-R3, R12, LR */
217 pop {r4} /* PC -> r4 */
218 pop {r5} /* xPSR -> r5 */
219 msr apsr_nzcvq, r5 /* Restore flags */
220
221 /* Set OS running flags */
222 ldr r5, =os_running
223 mov r6, #1
224 strb r6, [r5]
225
226 ldr r5, =running_task_count
227 strb r6, [r5]
228
229 /* Enable interrupts and jump to task */
230 cpsie i /* Enable IRQ */
231 bx r4 /* Branch to task entry point */
232
233.size start_cold_task, .-start_cold_task
