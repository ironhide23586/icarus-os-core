64void MPU_Config(void)
65{
66 MPU_Region_InitTypeDef r = {0};
67
68 HAL_MPU_Disable();
69
70 /* ---- Region 0: ITCM 64KB base — RO+exec for all ---- */
71 /* Kernel functions in ITCM - read-only prevents code modification */
72 /* Note: ARM MPU cannot prevent unprivileged EXECUTION of privileged code */
73 /* Protection relies on exception mechanism (SVC/PendSV only via instructions) */
74 r.Enable = MPU_REGION_ENABLE;
75 r.Number = MPU_REGION_ITCM_BASE;
76 r.BaseAddress = BSP_ITCM_BASE;
77 r.Size = MPU_REGION_SIZE_64KB;
78 r.AccessPermission = MPU_REGION_PRIV_RO_URO; /* Read-only for all */
79 r.IsBufferable = MPU_ACCESS_NOT_BUFFERABLE;
80 r.IsCacheable = MPU_ACCESS_NOT_CACHEABLE;
81 r.IsShareable = MPU_ACCESS_NOT_SHAREABLE;
82 r.DisableExec = MPU_INSTRUCTION_ACCESS_ENABLE;
83 r.TypeExtField = MPU_TEX_LEVEL1;
84 r.SubRegionDisable = 0x00;
85 HAL_MPU_ConfigRegion(&r);
86
87 /* ---- Region 1: QSPI 8MB flash — Priv+User RO+exec, cacheable ---- */
88 r.Enable = MPU_REGION_ENABLE;
89 r.Number = MPU_REGION_QSPI_FLASH;
90 r.BaseAddress = QSPI_BASE;
91 r.Size = MPU_REGION_SIZE_8MB;
92 r.AccessPermission = MPU_REGION_PRIV_RO_URO;
93 r.IsBufferable = MPU_ACCESS_BUFFERABLE;
94 r.IsCacheable = MPU_ACCESS_CACHEABLE;
95 r.IsShareable = MPU_ACCESS_NOT_SHAREABLE;
96 r.DisableExec = MPU_INSTRUCTION_ACCESS_ENABLE;
97 r.TypeExtField = MPU_TEX_LEVEL1;
98 r.SubRegionDisable = 0x00;
99 HAL_MPU_ConfigRegion(&r);
100
101 /* ---- Region 2: Internal Flash 128K — Priv+User RO+exec ---- */
102 r.Enable = MPU_REGION_ENABLE;
103 r.Number = MPU_REGION_FLASH;
104 r.BaseAddress = BSP_FLASH_BASE;
105 r.Size = MPU_REGION_SIZE_128KB;
106 r.AccessPermission = MPU_REGION_PRIV_RO_URO;
107 r.IsBufferable = MPU_ACCESS_NOT_BUFFERABLE;
108 r.IsCacheable = MPU_ACCESS_CACHEABLE;
109 r.IsShareable = MPU_ACCESS_NOT_SHAREABLE;
110 r.DisableExec = MPU_INSTRUCTION_ACCESS_ENABLE;
111 r.TypeExtField = MPU_TEX_LEVEL1;
112 r.SubRegionDisable = 0x00;
113 HAL_MPU_ConfigRegion(&r);
114
115 /* ---- Region 3: ITCM_PRIV — DISABLED (no additional protection) ---- */
116 /* Region 0 already provides RO+exec for all ITCM */
117 /* Kernel handlers (SVC, PendSV, SysTick) are protected by ARM arch: */
118 /* - SVC_Handler only invoked via 'svc' instruction */
119 /* - PendSV_Handler only invoked via PENDSVSET (privileged) */
120 /* - SysTick_Handler only invoked via SysTick interrupt */
121 /* Unprivileged code cannot directly call these handlers */
122 r.Enable = MPU_REGION_DISABLE;
123 r.Number = MPU_REGION_ITCM_PRIV;
124 r.BaseAddress = BSP_ITCM_BASE;
125 r.Size = MPU_REGION_SIZE_4KB;
126 r.AccessPermission = MPU_REGION_PRIV_RO_URO;
127 r.IsBufferable = MPU_ACCESS_NOT_BUFFERABLE;
128 r.IsCacheable = MPU_ACCESS_NOT_CACHEABLE;
129 r.IsShareable = MPU_ACCESS_NOT_SHAREABLE;
130 r.DisableExec = MPU_INSTRUCTION_ACCESS_ENABLE;
131 r.TypeExtField = MPU_TEX_LEVEL1;
132 r.SubRegionDisable = 0xE0;
133 HAL_MPU_ConfigRegion(&r);
134
135 /* ---- Region 4: Task data (dynamic, configured per context switch) ---- */
136 /* Left unconfigured here — set by MPU_ConfigureTaskData() */
137
138 /* ---- Region 5: DTCM 128K — PRIV_RW (DTCM protection enabled) ---- */
139 /* Kernel data (.dtcm_priv) lives here. */
140 /* Step 7: DTCM protection enabled - unprivileged tasks cannot access kernel data */
141 r.Enable = MPU_REGION_ENABLE;
142 r.Number = MPU_REGION_DTCM;
143 r.BaseAddress = BSP_DTCM_BASE;
144 r.Size = MPU_REGION_SIZE_128KB;
145 r.AccessPermission = MPU_REGION_PRIV_RW;
146 r.IsBufferable = MPU_ACCESS_NOT_BUFFERABLE;
147 r.IsCacheable = MPU_ACCESS_NOT_CACHEABLE;
148 r.IsShareable = MPU_ACCESS_NOT_SHAREABLE;
149 r.DisableExec = MPU_INSTRUCTION_ACCESS_DISABLE;
150 r.TypeExtField = MPU_TEX_LEVEL1;
151 r.SubRegionDisable = 0x00;
152 HAL_MPU_ConfigRegion(&r);
153
154 /* ---- Region 6: RAM_D1 512K — Priv+User Full Access ---- */
155 /* Stacks, .data, .bss, heap, task stacks/data pools. */
156 r.Enable = MPU_REGION_ENABLE;
157 r.Number = MPU_REGION_RAM_D1;
158 r.BaseAddress = BSP_RAM_D1_BASE;
159 r.Size = MPU_REGION_SIZE_512KB;
160 r.AccessPermission = MPU_REGION_FULL_ACCESS;
161 r.IsBufferable = MPU_ACCESS_BUFFERABLE;
162 r.IsCacheable = MPU_ACCESS_CACHEABLE;
163 r.IsShareable = MPU_ACCESS_NOT_SHAREABLE;
164 r.DisableExec = MPU_INSTRUCTION_ACCESS_DISABLE;
165 r.TypeExtField = MPU_TEX_LEVEL1;
166 r.SubRegionDisable = 0x00;
167 HAL_MPU_ConfigRegion(&r);
168
169 /* ---- Region 7: Peripherals 512MB — Priv+User Full Access, Device ---- */
170 /* GPIO, USB, SPI, I2C, UART, etc. */
171 r.Enable = MPU_REGION_ENABLE;
172 r.Number = MPU_REGION_PERIPH;
173 r.BaseAddress = BSP_PERIPH_BASE;
174 r.Size = MPU_REGION_SIZE_512MB;
175 r.AccessPermission = MPU_REGION_FULL_ACCESS;
176 r.IsBufferable = MPU_ACCESS_BUFFERABLE;
177 r.IsCacheable = MPU_ACCESS_NOT_CACHEABLE;
178 r.IsShareable = MPU_ACCESS_SHAREABLE;
179 r.DisableExec = MPU_INSTRUCTION_ACCESS_DISABLE;
180 r.TypeExtField = MPU_TEX_LEVEL0;
181 r.SubRegionDisable = 0x00;
182 HAL_MPU_ConfigRegion(&r);
183
184 /* Enable MPU with privileged default map as background fallback. */
185 /* MPU_PRIVILEGED_DEFAULT blocks unprivileged access to SCS (0xE000E000)*/
186 /* — NVIC, SCB, SysTick, MPU registers — forcing SVC for those writes. */
187 HAL_MPU_Enable(MPU_PRIVILEGED_DEFAULT);
188
189 /* Memory barriers required after MPU configuration (ARM Cortex-M7)
190 * DSB: Ensure all memory accesses complete before MPU change takes effect
191 * ISB: Flush instruction pipeline to ensure new MPU config is used */
192 __DSB();
193 __ISB();
194}
