54ITCM_FUNC bool __semaphore_feed(uint8_t semaphore_idx) {
55#ifdef HOST_TEST

57 if (semaphore_idx >= ICARUS_MAX_SEMAPHORES ||
58 !semaphore_list[semaphore_idx]->engaged) {
59 return false;
60 }

64 return false;
