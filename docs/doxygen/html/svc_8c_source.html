41void SVC_Handler_C(uint32_t *stack_frame) {
42 uint8_t svc_number = ((uint8_t *)(uintptr_t)stack_frame[6])[-2];
43 uint32_t arg0 = stack_frame[0];
44 uint32_t arg1 = stack_frame[1];
45
46 switch (svc_number) {
47
48 /* Critical section */
49 case SVC_ENTER_CRITICAL:
50 __enter_critical();
51 break;
52 case SVC_EXIT_CRITICAL:
53 __exit_critical();
54 break;
55
56 /* Scheduler */
57 case SVC_OS_YIELD:
58 __os_yield();
59 break;
60 case SVC_TASK_ACTIVE_SLEEP:
61 stack_frame[0] = __task_active_sleep(arg0);
62 break;
63 case SVC_OS_GET_TICK_COUNT:
64 stack_frame[0] = __os_get_tick_count();
65 break;
66 case SVC_OS_GET_CURRENT_TASK_NAME: {
67 const char *name = __os_get_current_task_name();
68 stack_frame[0] = (uint32_t)(uintptr_t)name;
69 break;
70 }
71 case SVC_OS_GET_RUNNING_TASK_COUNT:
72 stack_frame[0] = __os_get_running_task_count();
73 break;
74 case SVC_OS_GET_TASK_TICKS_REMAINING:
75 stack_frame[0] = __os_get_task_ticks_remaining();
76 break;
77
78 /* Task lifecycle */
79 case SVC_OS_REGISTER_TASK:
80 __os_register_task((void (*)(void))(uintptr_t)arg0,
81 (const char *)(uintptr_t)arg1);
82 break;
83 case SVC_OS_EXIT_TASK:
84 __os_exit_task();
85 break;
86 case SVC_OS_KILL_PROCESS:
87 __os_kill_process((uint8_t)arg0);
88 break;
89 case SVC_OS_TASK_SUICIDE:
90 __os_task_suicide();
91 break;
92
93 /* Kernel data */
94 case SVC_KERNEL_GET_STACK: {
95 uint32_t *ptr = __kernel_get_stack((uint8_t)arg0);
96 stack_frame[0] = (uint32_t)(uintptr_t)ptr;
97 break;
98 }
99 case SVC_KERNEL_GET_DATA: {
100 uint32_t *ptr = __kernel_get_data((uint8_t)arg0);
101 stack_frame[0] = (uint32_t)(uintptr_t)ptr;
102 break;
103 }
104 case SVC_KERNEL_PROTECTED_DATA: {
105 void *ptr = __kernel_protected_data((uint16_t)arg0);
106 stack_frame[0] = (uint32_t)(uintptr_t)ptr;
107 break;
108 }
109
110 /* Semaphore (non-spinning) */
111 case SVC_SEMAPHORE_INIT: {
112 bool ret = __semaphore_init((uint8_t)arg0, arg1);
113 stack_frame[0] = (uint32_t)ret;
114 break;
115 }
116 case SVC_SEMAPHORE_GET_COUNT:
117 stack_frame[0] = __semaphore_get_count((uint8_t)arg0);
118 break;
119 case SVC_SEMAPHORE_GET_MAX_COUNT:
120 stack_frame[0] = __semaphore_get_max_count((uint8_t)arg0);
121 break;
122
123 /* Pipe (non-spinning) */
124 case SVC_PIPE_INIT: {
125 bool ret = __pipe_init((uint8_t)arg0, (uint8_t)arg1);
126 stack_frame[0] = (uint32_t)ret;
127 break;
128 }
129 case SVC_PIPE_GET_COUNT:
130 stack_frame[0] = (uint32_t)__pipe_get_count((uint8_t)arg0);
131 break;
132 case SVC_PIPE_GET_MAX_COUNT:
133 stack_frame[0] = (uint32_t)__pipe_get_max_count((uint8_t)arg0);
134 break;
135
136 /* Spinning functions — not dispatched via SVC, run in thread mode */
137 case SVC_SEMAPHORE_FEED:
138 case SVC_SEMAPHORE_CONSUME:
139 case SVC_PIPE_ENQUEUE:
140 case SVC_PIPE_DEQUEUE:
141 case SVC_TASK_BUSY_WAIT:
142 case SVC_TASK_BLOCKING_SLEEP:
143 break;
144
145 /* Call gates for spinning functions — read kernel state atomically */
146 case SVC_SEM_CAN_FEED: {
147 bool r = __sem_can_feed((uint8_t)arg0);
148 stack_frame[0] = r ? 1u : 0u;
149 break;
150 }
151 case SVC_SEM_CAN_CONSUME: {
152 bool r = __sem_can_consume((uint8_t)arg0);
153 stack_frame[0] = r ? 1u : 0u;
154 break;
155 }
156 case SVC_PIPE_CAN_ENQUEUE: {
157 bool r = __pipe_can_enqueue((uint8_t)arg0, (uint8_t)arg1);
158 stack_frame[0] = r ? 1u : 0u;
159 break;
160 }
161 case SVC_PIPE_CAN_DEQUEUE: {
162 bool r = __pipe_can_dequeue((uint8_t)arg0, (uint8_t)arg1);
163 stack_frame[0] = r ? 1u : 0u;
164 break;
165 }
166
167 /* Write gates for spinning functions — modify kernel state atomically */
168 case SVC_SEM_INCREMENT:
169 __sem_increment((uint8_t)arg0);
170 break;
171 case SVC_SEM_DECREMENT:
172 __sem_decrement((uint8_t)arg0);
173 break;
174 case SVC_PIPE_WRITE_BYTES:
175 __pipe_write_bytes((uint8_t)arg0, (uint8_t *)(uintptr_t)arg1,
176 (uint8_t)stack_frame[2]);
177 break;
178 case SVC_PIPE_READ_BYTES:
179 __pipe_read_bytes((uint8_t)arg0, (uint8_t *)(uintptr_t)arg1,
180 (uint8_t)stack_frame[2]);
181 break;
182
183 /* Task metadata read gates — for display/diagnostics from unprivileged mode */
184 case SVC_GET_TASK_NAME: {
185 const char *name = __os_get_task_name((uint8_t)arg0);
186 stack_frame[0] = (uint32_t)(uintptr_t)name;
187 break;
188 }
189 case SVC_GET_NUM_TASKS:
190 stack_frame[0] = (uint32_t)__os_get_num_created_tasks();
191 break;
192 case SVC_OS_IS_RUNNING:
193 stack_frame[0] = (uint32_t)__os_is_running();
194 break;
195
196 default:
197 break;
198 }
199}
