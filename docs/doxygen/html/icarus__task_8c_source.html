442void os_init(void)
443{
444 uint8_t i;
445
446 /* Initialize scheduler state */
447 os_running = 0;
448 ticks_per_task = ICARUS_TICKS_PER_TASK;
449 running_task_count = 0;
450 current_task_index = 0;
451 current_cleanup_task_idx = -1;
452 current_task_ticks_remaining = ticks_per_task;
453
454 /* Initialize task pool */
455 for (i = 0; i < ICARUS_MAX_TASKS; i++) {
456 task_list[i] = &task_pool[i];
457 cleanup_task_idx[i] = -1;
458 }
459
460 /* Initialize semaphore pool */
461 for (i = 0; i < ICARUS_MAX_SEMAPHORES; i++) {
462 semaphore_list[i] = &semaphore_pool[i];
463 __init_sem(i, 0, false);
464 }
465
466 /* Initialize message pipe pool */
467 for (i = 0; i < ICARUS_MAX_MESSAGE_QUEUES; i++) {
468 message_pipe_list[i] = &message_pipe_pool[i];
469 __init_pipe(i, 0, false);
470 }
471
472 /* Clear virtual registers */
473 for (i = 0; i < 16; i++) {
474 cpu_vregisters[i] = 0;
475 }
476
477 /* Register system tasks */
478 os_register_task(os_idle_task, "ICARUS_KEEPALIVE_TASK");
479 os_register_task(os_heartbeat_task, ">ICARUS_HEARTBEAT<");
480
481 scheduler_enabled = true;
482}

787{

794 os_tick_count++;

941void os_kill_process(uint8_t task_index)
942{
943 if (task_index >= num_created_tasks) {
944 return;
945 }
946
947 /* Cannot kill system tasks, and task must be in active state */
948 if (task_index > 0 && task_index < num_created_tasks &&
949 task_list[task_index]->task_state < TASK_STATE_KILLED) {
950
951 printf("[INFO] %s killed %s\r\n",
952 task_list[current_task_index]->name,
953 task_list[task_index]->name);
954
955 task_list[task_index]->task_state = TASK_STATE_KILLED;
956
957 if (running_task_count > 0) {
958 running_task_count--;
959 }
960
961 if (current_cleanup_task_idx < (ICARUS_MAX_TASKS - 1)) {
962 cleanup_task_idx[++current_cleanup_task_idx] = (int8_t)task_index;
963 }
964
965 /* If killing self, yield immediately */
966 if (task_index == current_task_index) {
967 os_yield();
968 }
969 } else {
970 printf("[ERROR] Cannot kill task %d (state=%d)\r\n",
971 task_index, task_list[task_index]->task_state);
972 }
973}

1071void os_create_task(icarus_task_t *task, void (*function)(void),
1072 uint32_t *stack, uint32_t stack_size, const char *name)
1073{
1074 if (running_task_count >= ICARUS_MAX_TASKS) {
1075 return;
1076 }
1077
1078 /* Initialize TCB */
1079 task->function = function;
1080 task->stack_base = stack;
1081 task->stack_size = stack_size;
1082 task->task_state = TASK_STATE_COLD;
1083 task->task_priority = TASK_PRIORITY_LOW;
1084 task->global_tick_paused = 0;
1085 task->ticks_to_pause = 0;
1086
1087 /* Copy task name */
1088 strncpy(task->name, name, ICARUS_MAX_TASK_NAME_LEN);
1089 task->name[ICARUS_MAX_TASK_NAME_LEN - 1] = '\0';
1090
1091 /*
1092 * Initialize stack frame for first context switch
1093 * ARM Cortex-M exception frame (pushed automatically on exception):
1094 * [high addr] xPSR, PC, LR, R12, R3, R2, R1, R0 [low addr]
1095 *
1096 * We also reserve space for R4-R11 (saved by software in PendSV)
1097 */
1098 uint32_t *stack_top = stack + stack_size - 1;
1099
1100 /* Exception frame (hardware saved) */
1101 *(stack_top--) = 0x01000000; /* xPSR (Thumb bit set) */
1102 *(stack_top--) = (uint32_t)(uintptr_t)function; /* PC (entry point) */
1103 *(stack_top--) = (uint32_t)(uintptr_t)os_exit_task; /* LR (return addr) */
1104 *(stack_top--) = 0; /* R12 */
1105 *(stack_top--) = 0; /* R3 */
1106 *(stack_top--) = 0; /* R2 */
1107 *(stack_top--) = 0; /* R1 */
1108 *(stack_top) = 0; /* R0 <- SP points here */
1109
1110 task->stack_pointer = stack_top;
1111 num_created_tasks++;
1112}

1541ITCM_FUNC bool pipe_enqueue(uint8_t pipe_idx, uint8_t* message,
1542 uint8_t message_bytes)
1543{
1544 if (pipe_idx >= ICARUS_MAX_MESSAGE_QUEUES ||
1545 !message_pipe_list[pipe_idx]->engaged ||
1546 message_bytes > message_pipe_list[pipe_idx]->max_count) {
1547 return false;
1548 }
1549
1550 /* Block while insufficient space */
1551 while ((message_pipe_list[pipe_idx]->max_count -
1552 message_pipe_list[pipe_idx]->count) < message_bytes) {
1553 task_active_sleep(1);
1554 if ((message_pipe_list[pipe_idx]->max_count -
1555 message_pipe_list[pipe_idx]->count) >= message_bytes) {
1556 scheduler_enabled = false;
1557 }
1558 }
1559
1560 enter_critical();
1561
1562 /* Copy message bytes to circular buffer */
1563 for (uint8_t i = 0; i < message_bytes; i++) {
1564 message_pipe_list[pipe_idx]->buffer[
1565 message_pipe_list[pipe_idx]->enqueue_idx] = message[i];
1566 message_pipe_list[pipe_idx]->enqueue_idx =
1567 (uint8_t)(message_pipe_list[pipe_idx]->enqueue_idx + 1) %
1568 message_pipe_list[pipe_idx]->max_count;
1569 message_pipe_list[pipe_idx]->count++;
1570 }
1571
1572 message_pipe_list[pipe_idx]->tick_updated_at = os_tick_count;
1573 exit_critical();
1574
1575 return true;
