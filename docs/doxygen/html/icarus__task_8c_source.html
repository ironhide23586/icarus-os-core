313void os_init(void)
314{
315 uint8_t i;
316
317 /* Initialize scheduler state */
318 os_running = 0;
319 ticks_per_task = ICARUS_TICKS_PER_TASK;
320 running_task_count = 0;
321 current_task_index = 0;
322 current_cleanup_task_idx = -1;
323 current_task_ticks_remaining = ticks_per_task;
324
325 /* Initialize task pool */
326 for (i = 0; i < ICARUS_MAX_TASKS; i++) {
327 task_list[i] = &task_pool[i];
328 cleanup_task_idx[i] = -1;
329 }
330
331 /* Initialize semaphore pool */
332 for (i = 0; i < ICARUS_MAX_SEMAPHORES; i++) {
333 semaphore_list[i] = &semaphore_pool[i];
334 __init_sem(i, 0, false);
335 }
336
337 /* Initialize message pipe pool */
338 for (i = 0; i < ICARUS_MAX_MESSAGE_QUEUES; i++) {
339 message_pipe_list[i] = &message_pipe_pool[i];
340 __init_pipe(i, 0, false);
341 }
342
343 /* Clear virtual registers */
344 for (i = 0; i < 16; i++) {
345 cpu_vregisters[i] = 0;
346 }
347
348 /* Register system tasks */
349 os_register_task(os_idle_task, "ICARUS_KEEPALIVE_TASK");
350 os_register_task(os_heartbeat_task, ">ICARUS_HEARTBEAT<");
351
352 scheduler_enabled = true;
353}

464void os_kill_process(uint8_t task_index)
465{
466 if (task_index >= num_created_tasks) {
467 return;
468 }
469
470 /* Cannot kill system tasks, and task must be in active state */
471 if (task_index > 0 && task_index < num_created_tasks &&
472 task_list[task_index]->task_state < TASK_STATE_KILLED) {
473
474 printf("[INFO] %s killed %s\r\n",
475 task_list[current_task_index]->name,
476 task_list[task_index]->name);
477
478 task_list[task_index]->task_state = TASK_STATE_KILLED;
479
480 if (running_task_count > 0) {
481 running_task_count--;
482 }
483
484 if (current_cleanup_task_idx < (ICARUS_MAX_TASKS - 1)) {
485 cleanup_task_idx[++current_cleanup_task_idx] = (int8_t)task_index;
486 }
487
488 /* If killing self, yield immediately */
489 if (task_index == current_task_index) {
490 os_yield();
491 }
492 } else {
493 printf("[ERROR] Cannot kill task %d (state=%d)\r\n",
494 task_index, task_list[task_index]->task_state);
495 }
496}

509void os_create_task(icarus_task_t *task, void (*function)(void),
510 uint32_t *stack, uint32_t stack_size, const char *name)
511{
512 if (running_task_count >= ICARUS_MAX_TASKS) {
513 return;
514 }
515
516 /* Initialize TCB */
517 task->function = function;
518 task->stack_base = stack;
519 task->stack_size = stack_size;
520 task->task_state = TASK_STATE_COLD;
521 task->task_priority = TASK_PRIORITY_LOW;
522 task->global_tick_paused = 0;
523 task->ticks_to_pause = 0;
524
525 /* Copy task name */
526 strncpy(task->name, name, ICARUS_MAX_TASK_NAME_LEN);
527 task->name[ICARUS_MAX_TASK_NAME_LEN - 1] = '\0';
528
529 /*
530 * Initialize stack frame for first context switch
531 * ARM Cortex-M exception frame (pushed automatically on exception):
532 * [high addr] xPSR, PC, LR, R12, R3, R2, R1, R0 [low addr]
533 *
534 * We also reserve space for R4-R11 (saved by software in PendSV)
535 */
536 uint32_t *stack_top = stack + stack_size - 1;
537
538 /* Exception frame (hardware saved) */
539 *(stack_top--) = 0x01000000; /* xPSR (Thumb bit set) */
540 *(stack_top--) = (uint32_t)(uintptr_t)function; /* PC (entry point) */
541 *(stack_top--) = (uint32_t)(uintptr_t)os_exit_task; /* LR (return addr) */
542 *(stack_top--) = 0; /* R12 */
543 *(stack_top--) = 0; /* R3 */
544 *(stack_top--) = 0; /* R2 */
545 *(stack_top--) = 0; /* R1 */
546 *(stack_top) = 0; /* R0 <- SP points here */
547
548 task->stack_pointer = stack_top;
549 num_created_tasks++;
550}

660ITCM_FUNC bool pipe_enqueue(uint8_t pipe_idx, uint8_t* message,
661 uint8_t message_bytes)
662{
663 if (pipe_idx >= ICARUS_MAX_MESSAGE_QUEUES ||
664 !message_pipe_list[pipe_idx]->engaged ||
665 message_bytes > message_pipe_list[pipe_idx]->max_count) {
666 return false;
667 }
668
669 /* Block while insufficient space */
670 while ((message_pipe_list[pipe_idx]->max_count -
671 message_pipe_list[pipe_idx]->count) < message_bytes) {
672 task_active_sleep(1);
673 if ((message_pipe_list[pipe_idx]->max_count -
674 message_pipe_list[pipe_idx]->count) >= message_bytes) {
675 scheduler_enabled = false;
676 }
677 }
678
679 enter_critical();
680
681 /* Copy message bytes to circular buffer */
682 for (uint8_t i = 0; i < message_bytes; i++) {
683 message_pipe_list[pipe_idx]->buffer[
684 message_pipe_list[pipe_idx]->enqueue_idx] = message[i];
685 message_pipe_list[pipe_idx]->enqueue_idx =
686 (uint8_t)(message_pipe_list[pipe_idx]->enqueue_idx + 1) %
687 message_pipe_list[pipe_idx]->max_count;
688 message_pipe_list[pipe_idx]->count++;
689 }
690
691 message_pipe_list[pipe_idx]->tick_updated_at = os_tick_count;
692 exit_critical();
693
694 return true;
695}

697ITCM_FUNC bool pipe_dequeue(uint8_t pipe_idx, uint8_t* message,
698 uint8_t message_bytes)
699{
700 if (pipe_idx >= ICARUS_MAX_MESSAGE_QUEUES ||
701 !message_pipe_list[pipe_idx]->engaged ||
702 message_bytes > message_pipe_list[pipe_idx]->max_count) {
703 return false;
704 }
705
706 /* Block while insufficient data */
707 while (message_pipe_list[pipe_idx]->count < message_bytes) {
708 task_active_sleep(1);
709 if (message_pipe_list[pipe_idx]->count >= message_bytes) {
710 scheduler_enabled = false;
711 }
712 }
713
714 enter_critical();
715
716 /* Copy message bytes from circular buffer */
717 for (uint8_t i = 0; i < message_bytes; i++) {
718 message[i] = message_pipe_list[pipe_idx]->buffer[
719 message_pipe_list[pipe_idx]->dequeue_idx];
720 message_pipe_list[pipe_idx]->dequeue_idx =
721 (uint8_t)(message_pipe_list[pipe_idx]->dequeue_idx + 1) %
722 message_pipe_list[pipe_idx]->max_count;
723 message_pipe_list[pipe_idx]->count--;
724 }
725
726 message_pipe_list[pipe_idx]->tick_updated_at = os_tick_count;
727 exit_critical();
728
729 return true;
730}
