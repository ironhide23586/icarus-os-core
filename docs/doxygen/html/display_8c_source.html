125void msg_history_add(msg_history_t* hist, const uint8_t* data, uint8_t len,
126 uint8_t source_id, bool is_send) {
127 if (hist == NULL || data == NULL || len == 0) return;
128
129 // Clamp length to max
130 if (len > MSG_HISTORY_MAX_BYTES) {
131 len = MSG_HISTORY_MAX_BYTES;
132 }
133
134 // Add entry at head position
135 msg_history_entry_t* entry = &hist->entries[hist->head];
136 memcpy(entry->data, data, len);
137 entry->len = len;
138 entry->source_id = source_id;
139 entry->is_send = is_send;
140
141 // Advance head (circular)
142 hist->head = (uint8_t)(hist->head + 1) % MSG_HISTORY_LEN;
143 if (hist->count < MSG_HISTORY_LEN) {
144 hist->count++;
145 }
146}

169void display_render_msg_history(uint8_t row, uint8_t col, msg_history_t* hist, const char* label) {
170 if (hist == NULL) return;
171
172 // Header - use ASCII for compatibility
173 ANSI_GOTO(row, col);
174 printf(ANSI_CYAN ANSI_BOLD "+---%s---+" ANSI_RESET, label);
175
176 // Calculate starting index (oldest entry)
177 uint8_t start_idx;
178 if (hist->count < MSG_HISTORY_LEN) {
179 start_idx = 0;
180 } else {
181 start_idx = hist->head;
182 }
183
184 // Render each history entry
185 for (uint8_t i = 0; i < MSG_HISTORY_LEN; i++) {
186 ANSI_GOTO(row + 1 + i, col);
187
188 if (i < hist->count) {
189 uint8_t idx = (start_idx + i) % MSG_HISTORY_LEN;
190 msg_history_entry_t* entry = &hist->entries[idx];
191
192 if (entry->is_send) {
193 printf(ANSI_GREEN "|>P%d:", entry->source_id);
194 } else {
195 printf(ANSI_MAGENTA "|<C%d:", entry->source_id);
196 }
197
198 if (entry->len == 1) {
199 printf("%3d" ANSI_RESET "|", entry->data[0]);
200 } else {
201 for (uint8_t j = 0; j < entry->len && j < 3; j++) {
202 printf("%02X", entry->data[j]);
203 }
204 printf(ANSI_RESET "|");
205 }
206 } else {
207 printf("| |");
208 }
209 }
210
211 // Footer
212 ANSI_GOTO(row + MSG_HISTORY_LEN + 1, col);
213 printf(ANSI_CYAN "+---------+" ANSI_RESET);
214}

236void display_render_bar(uint8_t row, const char* task_name, uint32_t elapsed_ticks, uint32_t period_ticks) {
237 // Guard against division by zero
238 if (period_ticks == 0) {
239 period_ticks = 1;
240 }
241
242 // Clamp elapsed to period to avoid overflow
243 if (elapsed_ticks > period_ticks) {
244 elapsed_ticks = period_ticks;
245 }
246
247 // Calculate filled portion (0 to BAR_WIDTH)
248 // cppcheck-suppress zerodivcond
249 // period_ticks is guaranteed > 0 by guard above, but cppcheck can't track this
250 uint32_t filled = (elapsed_ticks * BAR_WIDTH) / period_ticks;
251 if (filled > BAR_WIDTH) {
252 filled = BAR_WIDTH;
253 }
254
255 // Move cursor to start of this task's row
256 ANSI_GOTO(row, 1);
257
258 // Print task name and opening bracket
259 printf("[%s] ", task_name ? task_name : "unknown");
260
261 // Render bar: filled portion, then empty portion
262 for (uint32_t i = 0; i < BAR_WIDTH; i++) {
263 if (i < filled) {
264 printf("█"); // Full block for filled
265 } else {
266 printf("─"); // Horizontal line for empty
267 }
268 }
269
270 // Print period in ticks (right-aligned)
271 printf(" %4" PRIu32 " / %4" PRIu32 " ticks", elapsed_ticks, period_ticks);
272
273 // Clear to end of line to avoid leftover characters
274 printf("\033[K");
275}
