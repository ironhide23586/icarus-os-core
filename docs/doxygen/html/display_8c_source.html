43void msg_history_add(msg_history_t* hist, const uint8_t* data, uint8_t len,
44 uint8_t source_id, bool is_send) {
45 if (hist == NULL || data == NULL || len == 0) return;
46
47 // Clamp length to max
48 if (len > MSG_HISTORY_MAX_BYTES) {
49 len = MSG_HISTORY_MAX_BYTES;
50 }
51
52 // Add entry at head position
53 msg_history_entry_t* entry = &hist->entries[hist->head];
54 memcpy(entry->data, data, len);
55 entry->len = len;
56 entry->source_id = source_id;
57 entry->is_send = is_send;
58
59 // Advance head (circular)
60 hist->head = (uint8_t)(hist->head + 1) % MSG_HISTORY_LEN;
61 if (hist->count < MSG_HISTORY_LEN) {
62 hist->count++;
63 }
64}

70void display_render_msg_history(uint8_t row, uint8_t col, msg_history_t* hist, const char* label) {
71 if (hist == NULL) return;
72
73 // Header - use ASCII for compatibility
74 ANSI_GOTO(row, col);
75 printf(ANSI_CYAN ANSI_BOLD "+---%s---+" ANSI_RESET, label);
76
77 // Calculate starting index (oldest entry)
78 uint8_t start_idx;
79 if (hist->count < MSG_HISTORY_LEN) {
80 start_idx = 0;
81 } else {
82 start_idx = hist->head;
83 }
84
85 // Render each history entry
86 for (uint8_t i = 0; i < MSG_HISTORY_LEN; i++) {
87 ANSI_GOTO(row + 1 + i, col);
88
89 if (i < hist->count) {
90 uint8_t idx = (start_idx + i) % MSG_HISTORY_LEN;
91 msg_history_entry_t* entry = &hist->entries[idx];
92
93 if (entry->is_send) {
94 printf(ANSI_GREEN "|>P%d:", entry->source_id);
95 } else {
96 printf(ANSI_MAGENTA "|<C%d:", entry->source_id);
97 }
98
99 if (entry->len == 1) {
100 printf("%3d" ANSI_RESET "|", entry->data[0]);
101 } else {
102 for (uint8_t j = 0; j < entry->len && j < 3; j++) {
103 printf("%02X", entry->data[j]);
104 }
105 printf(ANSI_RESET "|");
106 }
107 } else {
108 printf("| |");
109 }
110 }
111
112 // Footer
113 ANSI_GOTO(row + MSG_HISTORY_LEN + 1, col);
114 printf(ANSI_CYAN "+---------+" ANSI_RESET);
115}

124void display_render_bar(uint8_t row, const char* task_name, uint32_t elapsed_ticks, uint32_t period_ticks) {
125 // Guard against division by zero
126 if (period_ticks == 0) {
127 period_ticks = 1;
128 }
129
130 // Clamp elapsed to period to avoid overflow
131 if (elapsed_ticks > period_ticks) {
132 elapsed_ticks = period_ticks;
133 }
134
135 // Calculate filled portion (0 to BAR_WIDTH)
136 // cppcheck-suppress zerodivcond
137 // period_ticks is guaranteed > 0 by guard above, but cppcheck can't track this
138 uint32_t filled = (elapsed_ticks * BAR_WIDTH) / period_ticks;
139 if (filled > BAR_WIDTH) {
140 filled = BAR_WIDTH;
141 }
142
143 // Move cursor to start of this task's row
144 ANSI_GOTO(row, 1);
145
146 // Print task name and opening bracket
147 printf("[%s] ", task_name ? task_name : "unknown");
148
149 // Render bar: filled portion, then empty portion
150 for (uint32_t i = 0; i < BAR_WIDTH; i++) {
151 if (i < filled) {
152 printf("█"); // Full block for filled
153 } else {
154 printf("─"); // Horizontal line for empty
155 }
156 }
157
158 // Print period in ticks (right-aligned)
159 printf(" %4" PRIu32 " / %4" PRIu32 " ticks", elapsed_ticks, period_ticks);
160
161 // Clear to end of line to avoid leftover characters
162 printf("\033[K");
163}

195void display_render_vbar(uint8_t start_row, uint8_t col, uint32_t count, uint32_t max_count) {
196 if (max_count == 0) {
197 max_count = 1;
198 }
199 if (count > max_count) {
200 count = max_count;
201 }
202
203 uint32_t filled_rows = (count * VBAR_HEIGHT) / max_count;
204 if (filled_rows > VBAR_HEIGHT) {
205 filled_rows = VBAR_HEIGHT;
206 }
207
208 // Label
209 ANSI_GOTO(start_row - 1, col);
210 printf(ANSI_CYAN "SEM" ANSI_RESET);
211
212 // Top border
213 ANSI_GOTO(start_row, col);
214 printf("+---+");
215
216 // Bar rows (fill from bottom)
217 for (uint8_t i = 0; i < VBAR_HEIGHT; i++) {
218 ANSI_GOTO(start_row + 1 + i, col);
219 uint8_t row_from_bottom = VBAR_HEIGHT - 1 - i;
220 if (row_from_bottom < filled_rows) {
221 printf(ANSI_GREEN "|###|" ANSI_RESET);
222 } else {
223 printf("| |");
224 }
225 }
226
227 // Bottom border
228 ANSI_GOTO(start_row + VBAR_HEIGHT + 1, col);
229 printf("+---+");
230
231 // Count label
232 ANSI_GOTO(start_row + VBAR_HEIGHT + 2, col);
233 printf("%2" PRIu32 "/%2" PRIu32, count, max_count);
234}
