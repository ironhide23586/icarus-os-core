124void msg_history_add(msg_history_t* hist, const uint8_t* data, uint8_t len,
125 uint8_t source_id, bool is_send) {
126 if (hist == NULL || data == NULL || len == 0) return;
127
128 // Clamp length to max
129 if (len > MSG_HISTORY_MAX_BYTES) {
130 len = MSG_HISTORY_MAX_BYTES;
131 }
132
133 // Add entry at head position
134 msg_history_entry_t* entry = &hist->entries[hist->head];
135 memcpy(entry->data, data, len);
136 entry->len = len;
137 entry->source_id = source_id;
138 entry->is_send = is_send;
139
140 // Advance head (circular)
141 hist->head = (uint8_t)(hist->head + 1) % MSG_HISTORY_LEN;
142 if (hist->count < MSG_HISTORY_LEN) {
143 hist->count++;
144 }
145}

168void display_render_msg_history(uint8_t row, uint8_t col, msg_history_t* hist, const char* label) {
169 if (hist == NULL) return;
170
171 // Header - use ASCII for compatibility
172 ANSI_GOTO(row, col);
173 printf(ANSI_CYAN ANSI_BOLD "+---%s---+" ANSI_RESET, label);
174
175 // Calculate starting index (oldest entry)
176 uint8_t start_idx;
177 if (hist->count < MSG_HISTORY_LEN) {
178 start_idx = 0;
179 } else {
180 start_idx = hist->head;
181 }
182
183 // Render each history entry
184 for (uint8_t i = 0; i < MSG_HISTORY_LEN; i++) {
185 ANSI_GOTO(row + 1 + i, col);
186
187 if (i < hist->count) {
188 uint8_t idx = (start_idx + i) % MSG_HISTORY_LEN;
189 msg_history_entry_t* entry = &hist->entries[idx];
190
191 if (entry->is_send) {
192 printf(ANSI_GREEN "|>P%d:", entry->source_id);
193 } else {
194 printf(ANSI_MAGENTA "|<C%d:", entry->source_id);
195 }
196
197 if (entry->len == 1) {
198 printf("%3d" ANSI_RESET "|", entry->data[0]);
199 } else {
200 for (uint8_t j = 0; j < entry->len && j < 3; j++) {
201 printf("%02X", entry->data[j]);
202 }
203 printf(ANSI_RESET "|");
204 }
205 } else {
206 printf("| |");
207 }
208 }
209
210 // Footer
211 ANSI_GOTO(row + MSG_HISTORY_LEN + 1, col);
212 printf(ANSI_CYAN "+---------+" ANSI_RESET);
213}

235void display_render_bar(uint8_t row, const char* task_name, uint32_t elapsed_ticks, uint32_t period_ticks) {
236 // Guard against division by zero
237 if (period_ticks == 0) {
238 period_ticks = 1;
239 }
240
241 // Clamp elapsed to period to avoid overflow
242 if (elapsed_ticks > period_ticks) {
243 elapsed_ticks = period_ticks;
244 }
245
246 // Calculate filled portion (0 to BAR_WIDTH)
247 // cppcheck-suppress zerodivcond
248 // period_ticks is guaranteed > 0 by guard above, but cppcheck can't track this
249 uint32_t filled = (elapsed_ticks * BAR_WIDTH) / period_ticks;
250 if (filled > BAR_WIDTH) {
251 filled = BAR_WIDTH;
252 }
253
254 // Move cursor to start of this task's row
255 ANSI_GOTO(row, 1);
256
257 // Print task name and opening bracket
258 printf("[%s] ", task_name ? task_name : "unknown");
259
260 // Render bar: filled portion, then empty portion
261 for (uint32_t i = 0; i < BAR_WIDTH; i++) {
262 if (i < filled) {
263 printf("█"); // Full block for filled
264 } else {
265 printf("─"); // Horizontal line for empty
266 }
267 }
268
269 // Print period in ticks (right-aligned)
270 printf(" %4" PRIu32 " / %4" PRIu32 " ticks", elapsed_ticks, period_ticks);
271
272 // Clear to end of line to avoid leftover characters
273 printf("\033[K");
274}
